\begin{DoxyItemize}
\item {\ttfamily src/analytics/index.\+ts} -\/ Core analytics service with Litlyx integration

\begin{DoxyItemize}

\item {\ttfamily track\+Page\+View(path, data)} -\/ Track a page view
\item {\ttfamily track\+Button\+Click(button\+Id, data)} -\/ Track a button click
\item {\ttfamily track\+Form\+Submit(form\+Id, data)} -\/ Track a form submission

The following event types are predefined for consistency\+:


\begin{DoxyItemize}
\item {\ttfamily page\+\_\+view} -\/ Page views
\item {\ttfamily button\+\_\+click} -\/ Button clicks
\item {\ttfamily form\+\_\+submit} -\/ Form submissions
\item {\ttfamily user\+\_\+register} -\/ User registrations
\item {\ttfamily user\+\_\+login} -\/ User logins
\item {\ttfamily user\+\_\+logout} -\/ User logouts
\item {\ttfamily feature\+\_\+used} -\/ Feature usage
\item {\ttfamily error\+\_\+occurred} -\/ Errors
\end{DoxyItemize}
