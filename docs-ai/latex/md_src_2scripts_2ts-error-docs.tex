{\bfseries{Problem\+:}} Variables are defined with an underscore prefix ({\ttfamily \+\_\+variable}) but then used without the underscore ({\ttfamily variable}).

{\bfseries{Example\+:}}


\begin{DoxyCode}{0}
\DoxyCodeLine{//\ Error:}

\DoxyCodeLine{console.log(user.name)\ //\ Error:\ 'user'\ is\ not\ defined}
\DoxyCodeLine{}
\DoxyCodeLine{//\ Fixed:}
\DoxyCodeLine{const\ user\ =\ await\ getUser()}
\DoxyCodeLine{console.log(user.name)\ //\ Correctly\ using\ user}

{\bfseries{Example\+:}}

\DoxyCodeLine{}
\DoxyCodeLine{//\ Fixed:}
\DoxyCodeLine{const\ message:\ AIMessage\ =\ \{}
\DoxyCodeLine{\ \ role:\ 'user',}
\DoxyCodeLine{\ \ content:\ 'Hello',}
