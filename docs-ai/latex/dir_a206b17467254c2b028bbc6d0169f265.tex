\doxysubsection{Detailed Description}
Chardet is a character detection module for Node\+JS written in pure Javascript. Module is based on ICU project \href{http://site.icu-project.org/}{\texttt{ http\+://site.\+icu-\/project.\+org/}}, which uses character occurency analysis to determine the most probable encoding.\hypertarget{README.md_autotoc_md10092}{}\doxysubsection{\texorpdfstring{Installation}{Installation}}\label{README.md_autotoc_md10092}

\begin{DoxyCode}{0}
\DoxyCodeLine{npm\ i\ chardet}

\end{DoxyCode}
\hypertarget{README.md_autotoc_md10093}{}\doxysubsection{\texorpdfstring{Usage}{Usage}}\label{README.md_autotoc_md10093}
To return the encoding with the highest confidence\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{var\ chardet\ =\ require('chardet');}
\DoxyCodeLine{chardet.detect(Buffer.alloc('hello\ there!'));}
\DoxyCodeLine{//\ or}
\DoxyCodeLine{chardet.detectFile('/path/to/file',\ function(err,\ encoding)\ \{\});}
\DoxyCodeLine{//\ or}
\DoxyCodeLine{chardet.detectFileSync('/path/to/file');}

\end{DoxyCode}

\begin{DoxyCode}{0}

\DoxyCodeLine{//\ or}
\DoxyCodeLine{chardet.detectFileAll('/path/to/file',\ function(err,\ encoding)\ \{\});}
\DoxyCodeLine{//\ or}
\DoxyCodeLine{chardet.detectFileAllSync('/path/to/file');}
