Color data

A data object that includes color data.

ts
type Color = ColorData;
interface ColorData {
model: AcceptedModel; // 'rgb'|'hwb'|'hsl'|'hsv'|'cmyk'
values: number[]; // e.g. if rgb: [255, 255, 255], if cmyk: [100, 100, 0, 100]
alpha?: number; // The opacity value from 0 to 1.
}
